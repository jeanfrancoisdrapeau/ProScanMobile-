885  {
886  string[] suf = { "B", "KB", "MB", "GB", "TB", "PB", "EB" };
887  if (byteCount == 0)
888  return "0" + suf[0];
889  long bytes = Math.Abs(byteCount);
890  int place = Convert.ToInt32(Math.Floor(Math.Log(bytes, 1024)));
891  double num = Math.Round(bytes / Math.Pow(1024, place), 1);
892  return string.Format("{0:#0.0}", (Math.Sign(byteCount) * num)) + suf[place];
893  }
